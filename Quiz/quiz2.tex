				      v_1 \\ v_2 \\ v_3
			      \end{pmatrix} \,,
			      \vec{w} = \begin{pmatrix}
				      w_1 \\ w_2 \\ w_3
			      \end{pmatrix} \,?
		      \end{equation*}

		\item Let $\mathbf{r}(t)$ be a curve in $\mathbb{R}^3$. Let $\mathbf{T}(t_0)$ be the unit tangent vector of the curve at the point $\mathbf{r}(t_0)$.
		      What can you say about $\mathbf{T}(t_0) \cdot \mathbf{T}'(t_0)$? Why?
	\end{enumerate}
