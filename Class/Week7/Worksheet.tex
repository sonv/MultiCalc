\begin{question}
Compute directional derivative in the direction $\vect{u} \not= \vect{0}$ of the functions
\begin{enumerate}
    \item $f(\vect{x}) = |\vect{x}|^2$.
    \item $$ f(\vect{x}) = 
        \begin{cases}
            \frac{x_1^2 x_2}{x_1^4 + x_2^2} & \vect{x} \not= 0 \\
            0 & \vect{x} = 0 \,.
        \end{cases}$$
\end{enumerate}
\end{question}
