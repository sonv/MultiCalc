\begin{problem}
 Consider the function $f(x, y)=x^4 y^2-x y^3-8 x+8 y+13$.
\begin{enumerate}
    \item  Show that $f$ has a critical point at $x=1, y=2$.
    \item Compute the Hessian $H$.
    \item Classify the critical point at $(1,2)$.
\end{enumerate}

\end{problem}
